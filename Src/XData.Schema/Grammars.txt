

compilation-unit:
uri-aliasing* namespace*
;

uri-aliasing:
'alias' uri-value 'as' uri-alias ';'
;
uri-value:
string-value
;
uri-alias:
name
;
uri:
uri-value | uri-alias
;

namespace:
'namespace' uri '{' namespace-import* namespace-member* '}'
;
namespace-import:
'import' uri ('as' namespace-alias)? //';'
;
namespace-alias:
name
;
namespace-member:
type | global-attribute | global-element 
;

qualifiable-name:
(namespace-alias ':')? name
;

type:
'type' name
;
